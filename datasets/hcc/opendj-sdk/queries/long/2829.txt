Set up 2 servers and enable replication:

$ ./DJ1/opendj/setup -h localhost -p 1389 -D "cn=myself" -w password --adminConnectorPort 4444 -Z 1636 --generateSelfSignedCertificate -b dc=com -t je --cli -n
$ ./DJ2/opendj/setup -h localhost -p 1390 -D "cn=myself" -w password --adminConnectorPort 4445 -Z 1637 --generateSelfSignedCertificate -b dc=com -t je  --cli -n
$ ./DJ1/opendj/bin/dsreplication enable --host1 localhost --port1 4444 --bindDN1 "cn=myself" --bindPassword1 "password" --replicationPort1 8989 --host2 localhost --port2 4445 --bindDN2 "cn=myself" --bindPassword2 "password" --replicationPort2 8990 -b dc=com -I admin -w password  -X -n
$ ./DJ1/opendj/bin/dsreplication initialize-all -h localhost -p 4444 -b dc=com -I admin -w password -X -n

$ ./DJ1/opendj/bin/stop-ds
