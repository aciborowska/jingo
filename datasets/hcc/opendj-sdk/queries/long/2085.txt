1. configure an instance with pdb backend and some automatically-generated data

$ ./opendj/setup --cli -n -p 1389 --adminConnectorPort 1444 -b "dc=com" -D "cn=myself" -w "password" -t pdb -d 160


2. create a second pdb backend

$ ./opendj/bin/dsconfig create-backend --backend-name "nonDefault" --set base-dn:"dc=non-default,dc=com" --set enabled:true --set writability-mode:enabled -t pdb --set db-cache-percent:5 -p 1444 -D "cn=myself" -w "password" -n -X


3. stop the server

$ ./opendj/bin/stop-ds


4. export the data from the first backend

$ ./opendj/bin/export-ldif -l ./export.ldif -n userRoot


5. re-import data in the first backend

$ ./opendj/bin/import-ldif -l ./export.ldif -n userRoot
