            "userpassword: changeit",
            "objectclass: inetorgperson",
            "objectclass: organizationalPerson",
            "objectclass: top",
            "objectclass: person",
            "l: DemoLand"));
    Connection myconnection = Connections.newInternalConnection(memoryBackend);
