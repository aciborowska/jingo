As reported in #2679 the current Accept header sent by Google Chrome forces a HTTP 500, again.
> eXist 5.1.0-SNAPSHOT 

### Reproduce
```bash
docker run -p 8099:8080 existdb/existdb:latest & sleep 30;
while [ $(curl --head --silent http://localhost:8099/ --connect-timeout 5 | grep -c "200 OK") = 0 ]; do sleep 2s; done;
echo 'xquery version "3.1";
module namespace api = "http://test.org/ns/exist/api";
declare namespace output = "http://www.w3.org/2010/xslt-xquery-serialization";
declare namespace rest = "http://exquery.org/ns/restxq";

(:~
 : Test
 :)
declare
  %rest:GET
  %rest:path("/info")
  %rest:produces("application/json")
  %output:media-type("application/json")
  %output:method("json")
function api:info()
as map(*) {
  map {
    "name": "test"
  }
};' > /tmp/test.xqm
curl -X PUT -H 'Content-Type: application/xquery' --data-binary @/tmp/test.xqm http://admin:@localhost:8099/exist/rest/db/test.xqm
# google-chrome http://localhost:8099/exist/restxq/info
# or in the terminal:
curl -H "Accept: text/html,application/xhtml+xml,application/xml;q=0.9,image/webp,image/apng,*/*;q=0.8,application/signed-exchange;v=b3;q=0.9" http://localhost:8099/exist/restxq/info
```
