position() on reverse axes goes by forward document order, but should count backwards, as discovered by @gioele.

http://markmail.org/thread/ijvbtkejott2bwj7

```
xquery version "3.0";

let $doc := 
    <doc>
        <entry>a</entry>
        <entry>b</entry>
        <entry>c</entry>
    </doc>

let $preceding-siblings-a := $doc//entry[. eq "c"]/preceding-sibling::entry[position() = 1] (:incorrect:)
let $preceding-siblings-b := $doc//entry[. eq "c"]/preceding-sibling::entry[1] (:correct:)
let $preceding-siblings-c := $doc//entry[. eq "c"]/preceding-sibling::entry[last()] (:correct:)

return 
    <result>{$preceding-siblings-a, $preceding-siblings-b, $preceding-siblings-c}</result>
```
