### What is the problem

In a search application I'm building, I encountered an NPE when using `util:expand` to expand hits in a query that includes hits on Lucene full text and ngram indexes. 

### What did you expect

[The documentation](https://exist-db.org/exist/apps/doc/kwic) indicates that Lucene full text and ngram indexes both support match highlighting, so it should be possible to apply `util:expand` to a node that contains hits from both index types. Doing so should certainly not cause an NPE.

### Describe how to reproduce or add a test

The following query will reproduce the error, in the form of 3 tests wrapped in try-catch blocks. While test 1 is correct, tests 2 and 3 trigger an NPE:

```xquery
xquery version "3.1";

(: setup :)
let $data := 
    <root>
        <x>Hello</x>
        <y>Hello</y>
    </root>
let $xconf := 
    <collection xmlns="http://exist-db.org/collection-config/1.0">
        <index>
            <lucene>
                <text qname="x"/>
            </lucene>
            <ngram qname="y"/>
        </index>
    </collection>
let $xconf-col := xmldb:create-collection("/db/system/config/db", "temp")
let $store-xconf := xmldb:store($xconf-col, "collection.xconf", $xconf)
let $data-col := xmldb:create-collection("/db", "temp")
let $store-data := xmldb:store($data-col, "test.xml", $data)

(: the heart of the test :)
let $q := "Hello"
let $doc := doc($store-data)
let $x-hits := $doc/root[ft:query(x, $q)]
let $y-hits := $doc/root[ngram:contains(y, $q)]
let $hits := ($x-hits | $y-hits)
return
    <tests>
        <test n="1">{ try { util:expand($x-hits) } catch * { $err:code } }</test>
        <test n="2">{ try { util:expand($y-hits) } catch * { $err:code } }</test>
        <test n="3">{ try { util:expand($hits)   } catch * { $err:code } }</test>
    </tests>
```

The results I would have expected:

```xml

    </test>
    <test n="2">
        <root>
            <x>Hello</x>
            <y><exist:match xmlns:exist="http://exist.sourceforge.net/NS/exist">Hello</exist:match></y>
        </root>
    </test>
    <test n="3">
        <root>
            <x><exist:match xmlns:exist="http://exist.sourceforge.net/NS/exist">Hello</exist:match></x>
            <y><exist:match xmlns:exist="http://exist.sourceforge.net/NS/exist">Hello</exist:match></y>
        </root>
    </test>
</tests>
```
