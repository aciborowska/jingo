     * @param args
     */
    public static void main(String[] args) {
        HazelcastInstance hci = Hazelcast.newHazelcastInstance();

        IMap<String, byte[]> map = hci.getMap("cache");
        for(int i=0;i<500;i++)
        {
            map.set(String.valueOf(i), new byte[500]);
        }

        for(int repetitions=0;repetitions < 10;repetitions++)
        {
            System.out.println("---------------New Iteration---------------");
            for(int i=0;i<500;i++)
            {
                byte[] data = map.get(String.valueOf(i));
                if(data != null)
                {
                    System.out.println("ok");
                }
                else
                {
                    System.out.println("NOT FOUND");
                }
            }
        }

        Hazelcast.shutdownAll();
    }

}
