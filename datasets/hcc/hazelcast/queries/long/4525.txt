```
public class HazelcastIndexTest {

    public static class TestClass implements Serializable {
        private String field;

        public TestClass(final String field) {
            this.field = field;
        }

        public String getField() {
            return field;
        }

    }

    @Test
    public void hazelcastIndexDoesntWork() {
        final HazelcastInstance hazelcast = Hazelcast.newHazelcastInstance(null);
        final IMap<String, TestClass> map = hazelcast.getMap("map");

        // given
        map.addIndex("field", false); // if we remove this line, test will pass
        map.put("1000", new TestClass(null));
        map.put("1001", new TestClass("not null"));

        // when
        final Predicate emptyFieldPredicate = Predicates.notEqual("field", "value");
        final Collection<TestClass> emptyFieldValues = map.values(emptyFieldPredicate);

        // then
        assertThat(emptyFieldValues, hasSize(2));
    }

}
```

Apparently test should return both of the entries but it ignores one with "null" value.
If we remove index for "field", it will pass.
