Destroying a replicated map produces a `CLEAR_ALL` map event that is not generated for IMap or MultiMap

Reproducer

```
import com.hazelcast.core.EntryEvent;
import com.hazelcast.core.EntryListener;
import com.hazelcast.core.Hazelcast;
import com.hazelcast.core.HazelcastInstance;
import com.hazelcast.map.IMap;
import com.hazelcast.map.MapEvent;
import com.hazelcast.multimap.MultiMap;
import com.hazelcast.replicatedmap.ReplicatedMap;

@SuppressWarnings("rawtypes")
public class Application implements EntryListener {

    public static void main(String[] args) {
        Application application = new Application();
        application.run();
    }
    
    @SuppressWarnings("unchecked")
    void run() {
        HazelcastInstance hazelcastInstance = Hazelcast.newHazelcastInstance();
        
        IMap iMap = hazelcastInstance.getMap("iMap");
        MultiMap multiMap = hazelcastInstance.getMultiMap("multiMap");
        ReplicatedMap replicatedMap = hazelcastInstance.getReplicatedMap("replicatedMap");

        iMap.addEntryListener(this, true);
        multiMap.addEntryListener(this, true);
        replicatedMap.addEntryListener(this);
        
        iMap.destroy();
        multiMap.destroy();
        replicatedMap.destroy();
        
        hazelcastInstance.shutdown();
    }

    @Override
    public void entryAdded(EntryEvent event) {
    }

    @Override
    public void entryUpdated(EntryEvent event) {
    }

    @Override
    public void entryRemoved(EntryEvent event) {
    }

    @Override
    public void entryEvicted(EntryEvent event) {
    }

    @Override
    public void entryExpired(EntryEvent event) {
    }

    @Override
    public void mapCleared(MapEvent event) {
        System.out.println(event);
    }

    @Override
    public void mapEvicted(MapEvent event) {
    }


}
```
