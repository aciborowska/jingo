  <xs:element name="queue">
        xs:complexType
            xs:sequence
                <xs:element ref="max-size-per-jvm" minOccurs="0" maxOccurs="1"/>
                <xs:element ref="backing-map-ref" minOccurs="0" maxOccurs="1"/>
                <xs:element ref="item-listeners" minOccurs="0" maxOccurs="1"/>
            /xs:sequence
            <xs:attribute name="name" use="optional" default="default">
                xs:simpleType
                    <xs:restriction base="xs:string"/>
                /xs:simpleType
            /xs:attribute
        /xs:complexType
    /xs:element
