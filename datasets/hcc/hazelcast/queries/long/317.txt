the executor xsd element also requires a lot of elements to be available. Afaik it would be better to make them all optional:

  <xs:element name="executor-service">
        xs:complexType
            xs:sequence
                <xs:element ref="core-pool-size" minOccurs="0" maxOccurs="1"/>
                <xs:element ref="max-pool-size" minOccurs="0" maxOccurs="1"/>
                <xs:element ref="keep-alive-seconds" minOccurs="0" maxOccurs="1"/>
            /xs:sequence
            <xs:attribute name="name" use="optional" default="default">
                xs:simpleType
                    <xs:restriction base="xs:string"/>
                /xs:simpleType
            /xs:attribute
        /xs:complexType
    /xs:element
