`ScheduledTaskHandlerImpl.of(urn)` assumes address is IPv4.

```java
    public static void main(String[] args) throws Exception {
        setIPv6Address();

        HazelcastInstance instance = Hazelcast.newHazelcastInstance();
        IScheduledExecutorService scheduler = instance.getScheduledExecutorService("scheduler");
        IScheduledFuture<?> future = scheduler.scheduleOnMember(new Runnable() {
            @Override
            public void run() {
            }
        }, instance.getCluster().getLocalMember(), 1, TimeUnit.HOURS);

        String urn = future.getHandler().toUrn();
        future = scheduler.getScheduledFuture(ScheduledTaskHandler.of(urn));
    }

    private static void setIPv6Address() throws SocketException {
        Enumeration<NetworkInterface> networkInterfaces = NetworkInterface.getNetworkInterfaces();
        while (networkInterfaces.hasMoreElements()) {
            NetworkInterface networkInterface = networkInterfaces.nextElement();
            Enumeration<InetAddress> inetAddresses = networkInterface.getInetAddresses();
            while (inetAddresses.hasMoreElements()) {
                InetAddress address = inetAddresses.nextElement();
                if (!address.isLoopbackAddress() && address instanceof Inet6Address) {
                    System.setProperty("hazelcast.local.localAddress", address.getHostAddress());
                    return;
                }
            }
        }
    }
```
