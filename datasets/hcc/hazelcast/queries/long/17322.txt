The following configuration for a client integrated with Spring get's an error for `Invalid element name: native-memory`:

```
<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"
    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
    xmlns:hz="http://www.hazelcast.com/schema/spring" xsi:schemaLocation="
http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd
http://www.hazelcast.com/schema/spring http://www.hazelcast.com/schema/spring/hazelcast-spring.xsd">

    <hz:client id="hzInstance">

        <hz:near-cache name="sample-store" in-memory-format="NATIVE">
            <hz:preloader enabled="true" directory="..." store-interval-seconds="86400" store-initial-delay-seconds="60"/>
        </hz:near-cache>
        <hz:native-memory enabled="true" metadata-space-percentage="12.5" allocator-type="POOLED" min-block-size="16">
            <hz:size value="512" unit="MEGABYTES"/>
        </hz:native-memory>
    </hz:client>
</beans>
```

![image](https://user-images.githubusercontent.com/20101711/89686127-bf707380-d8cb-11ea-835c-be916d31dd0e.png)

The equivalent 'non-Spring' configuration is considered valid:

```
<hazelcast-client xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                  xsi:schemaLocation="http://www.hazelcast.com/schema/client-config
                               http://www.hazelcast.com/schema/client-config/hazelcast-client-config.xsd"
                  xmlns="http://www.hazelcast.com/schema/client-config">

    <near-cache name="sample-store">
        <in-memory-format>NATIVE</in-memory-format>
        <preloader enabled="true" directory="..." store-interval-seconds="600" store-initial-delay-seconds="60"/>
    </near-cache>

    <native-memory enabled="true" allocator-type="POOLED">
        <metadata-space-percentage>12.5</metadata-space-percentage>
        <min-block-size>16</min-block-size>
        <size value="512" unit="MEGABYTES"/>
    </native-memory>

</hazelcast-client>
```

This can be observed with older versions (such as 3.9) as well as the latest version.
