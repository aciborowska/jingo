In 3.12.3, a `ClientConfig` with Kubernetes discovery has this discovery section lost when passing through ClientConfigXmlGenerator.

Reproducer:

```
import java.io.ByteArrayInputStream;
import java.io.IOException;
import java.nio.charset.StandardCharsets;
import java.util.List;

import com.hazelcast.client.config.ClientAliasedDiscoveryConfigUtils;
import com.hazelcast.client.config.ClientConfig;
import com.hazelcast.client.config.ClientConfigXmlGenerator;
import com.hazelcast.client.config.XmlClientConfigBuilder;
import com.hazelcast.config.AliasedDiscoveryConfig;
import com.hazelcast.config.KubernetesConfig;

public class Main {

	public static void main(String[] args) throws IOException {
		ClientConfig clientConfigBefore = new ClientConfig();
		
		KubernetesConfig kubernetesConfig = new KubernetesConfig();
		kubernetesConfig.setEnabled(true);
		kubernetesConfig.setProperty("service-dns", "kubernetes");
		
		clientConfigBefore.getNetworkConfig().setKubernetesConfig(kubernetesConfig);
		
		System.out.println("Before: " + clientConfigBefore.getNetworkConfig().getKubernetesConfig());
		
		String xmlString = ClientConfigXmlGenerator.generate(clientConfigBefore);

		ByteArrayInputStream inputStream = new ByteArrayInputStream(xmlString.getBytes(StandardCharsets.UTF_8));
        ClientConfig clientConfigAfter = new XmlClientConfigBuilder(inputStream).build();
		
		List<AliasedDiscoveryConfig<?>> aliasedDiscoveryConfigs =
				ClientAliasedDiscoveryConfigUtils.aliasedDiscoveryConfigsFrom(clientConfigAfter);
		for (AliasedDiscoveryConfig<?> aliasedDiscoveryConfig : aliasedDiscoveryConfigs) {
			if (aliasedDiscoveryConfig.getTag().equals("kubernetes")) {
				System.out.println(aliasedDiscoveryConfig.toString());				
			}
		}
	}
}
```
