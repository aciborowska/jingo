public void testAddAll_whenCollectionContainsNull() {
    IQueue<String> queue = getQueue();
    for (int i = 0; i < 10; i++) {
        queue.offer("item" + i);
    }
    List<String> list = new ArrayList<String>();
    list.add("item10");
    list.add(null);

    queue.addAll(list);
    assertEquals(12, queue.size());
}

}
