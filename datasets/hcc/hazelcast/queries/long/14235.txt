Reproducer, for 3.11

```
import com.hazelcast.client.HazelcastClient;
import com.hazelcast.core.Hazelcast;
import com.hazelcast.core.HazelcastInstance;

public class Main {
	
	private static class World implements java.io.Serializable {
		private static final long serialVersionUID = 1L;
	}
	
    public static void main(String[] args) {
    	HazelcastInstance server = Hazelcast.newHazelcastInstance();
    	HazelcastInstance client1 = HazelcastClient.newHazelcastClient();
    	HazelcastInstance client2 = HazelcastClient.newHazelcastClient();
    	World world = new World();
    	client1.getReplicatedMap("etwas").put("hello", world);
    	System.out.println("Before get()");
    	System.out.println(client2.getReplicatedMap("etwas").get("hello"));
    	System.out.println("Before remove()");
    	System.out.println(client2.getReplicatedMap("etwas").remove("hello"));
    	System.out.println("Before shuwdown()");
    	client1.shutdown();
    	client2.shutdown();
    	server.shutdown();
    }
}
```

Note the `get()` works, but the `remove()` throws an exception.

See [StackOverflow ticket 53652395](https://stackoverflow.com/questions/53652395/hazelcast-classnotfounderror-on-dedicated-member-for-hazelclient-replicated-map)
