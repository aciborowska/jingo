public class Main {
    public static void main(String[] args) {
//        String host = "splja01w7";
        String host = "localhost";

        Config cfg = new Config();
        cfg.setProperty("hazelcast.jmx.detailed", "true");
        cfg.setProperty("hazelcast.jmx", "true");
        cfg.setProperty("hazelcast.prefer.ipv4.stack", "true");
        cfg.setProperty("hazelcast.executor.client.thread.count", "20");
        cfg.setProperty("hazelcast.version.check.enabled", "false");


        NetworkConfig network = cfg.getNetworkConfig();
        Join join = network.getJoin();
        join.getMulticastConfig().setEnabled(false);

        TcpIpConfig tcpConfig = join.getTcpIpConfig();
        tcpConfig.addMember(host+":7004");
        tcpConfig.setEnabled(true);

        network.setPortAutoIncrement(false);
        network.setPort(7004);

        cfg.setProperty("hazelcast.local.localAddress", host);
        cfg.getGroupConfig().setName("dev").setPassword("dev-pass");

        HazelcastInstance instance = Hazelcast.newHazelcastInstance(cfg);
        new Thread(new MyRunnable(instance)).start();

    }

    public static class MyRunnable implements Runnable{
        private HazelcastInstance instance;

        public MyRunnable(HazelcastInstance instance) {
            this.instance = instance;
        }

        @Override
        public void run() {
            while (true) {
                try {
                    for (Client c : instance.getClientService().getConnectedClients()){
                        System.out.println(c.getSocketAddress());
                    }
                } catch(Exception e) {
                    e.printStackTrace();
                }

                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}

public class MainClient {
