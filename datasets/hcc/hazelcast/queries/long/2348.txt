with the following code:

```
public static void main(String[] args) throws InterruptedException {
    int THREADS_NUMBER = 1;

    HazelcastInstance instance = Hazelcast.newHazelcastInstance();
    List<Thread> threads =
            Stream.iterate(0, i -> i + 1).limit(THREADS_NUMBER).map(i -> processingThread(instance)).collect(Collectors.toList());

    threads.forEach(Thread::start);
    threads.forEach(Uninterruptibles::joinUninterruptibly);
    Hazelcast.shutdownAll();
}

private static Thread processingThread(HazelcastInstance instance) {
    return new Thread(() -> {
        ILock lock = instance.getLock("lock");
        while (true) {
            lock.lock();
            try {
                TransactionOptions options = new TransactionOptions().setTransactionType(TransactionOptions.TransactionType.LOCAL);
                TransactionContext transactionContext = instance.newTransactionContext(options);
                transactionContext.beginTransaction();

                TransactionalQueue<Integer> queue = transactionContext.getQueue("queue");
                Stream.iterate(0, i -> i + 1).limit(100).forEach(queue::offer);
                Stream.iterate(0, i -> i + 1).limit(50).forEach(i -> queue.poll());

                transactionContext.commitTransaction();
            } finally {
                lock.unlock();
            }
            Uninterruptibles.sleepUninterruptibly(1, TimeUnit.MILLISECONDS);
        }
    });
}
```
