A = load 'filterbug.data' using PigStorage() as ( id, str );

A = foreach A generate
        id,
        str,
        (
        str matches 'hello' or
        str matches 'hello'
        ? 1 : 0
        )                       as matched;
describe A;

B = group A by ( id );
describe B;

D = foreach B generate
        group,
        SUM(A.matched)  as matchedcount,
        A;
describe D;

E = filter D by matchedcount > 0;
describe E;

F = foreach E generate
        FLATTEN(A);
